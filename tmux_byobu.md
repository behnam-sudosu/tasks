#tmux
	
##tmux help you if disconnect from server
	
	
	sudo apt install tmux
	tmux
	<!-- after log out -->
	tmux ls
	<!-- you can see you tmux session -->
	tmux att -t (number of session)
	<!-- close tmux -->
	ctrl + d
